def java_class
  names = self.name.split('::')
  package = Java
  names[0..-2].each do |module_name|
    package = package.__send__(module_name.downcase)
  end

  package.__send__(names[-1])
end

def javaify(arg)
  if Array === arg
    arg.map{|a| javaify(a)}
  else
    case(arg)
    when Regexp
      java.util.regex.Pattern.compile(arg.source)
    when Symbol
      arg.to_s
    when IO, StringIO
      IOWriter.new(arg)
    else
      arg
    end
  end
end
