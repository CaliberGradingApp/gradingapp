def add_callback(*names, &block)
  names.each do |name|
    callbacks[name] ||= []
    callbacks[name] << block
  end
end

def node_end(node)
  @node = node
  self.class.get_callbacks("end_#{node.sexp_type}".to_sym).each do |block|
    self.instance_exec(node, &block)
  end
end
