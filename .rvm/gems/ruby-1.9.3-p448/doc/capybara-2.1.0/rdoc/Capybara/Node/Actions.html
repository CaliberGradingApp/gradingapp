def attach_file(locator, path, options={})
  Array(path).each do |p|
    raise Capybara::FileNotFound, "cannot attach file, #{p} does not exist" unless File.exist?(p.to_s)
  end
  find(:file_field, locator, options).set(path)
end

    @param [String] locator Which field to fill in @param [Hash{:with => String}] options The value to fill in

    # File lib/capybara/node/actions.rb, line 50
def fill_in(locator, options={})
  raise "Must pass a hash containing 'with'" if not options.is_a?(Hash) or not options.has_key?(:with)
  with = options.delete(:with)
  find(:fillable_field, locator, options).set(with)
end

    page.select 'March', :from => 'Month'

    @param [String] value Which option to select @param [Hash{:from => String}] options The id, name or label of the select box

    # File lib/capybara/node/actions.rb, line 106
def select(value, options={})
  if options.has_key?(:from)
    from = options.delete(:from)
    find(:select, from, options).find(:option, value, options).select_option
  else
    find(:option, value, options).select_option
  end
end
