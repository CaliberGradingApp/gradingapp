def [](key)
  if value = super
    value
  elsif @default_proc && !key?(key)
    @default_proc.call(self, key)
  else
    value
  end
end

def fetch(key, default_value = NULL)
  if NULL != (value = get_or_default(key, NULL))
    value
  elsif block_given?
    yield key
  elsif NULL != default_value
    default_value
  else
    raise KEY_ERROR, 'key not found'
  end
end

def marshal_dump

  each_pair {|k, v| h[k] = v}
