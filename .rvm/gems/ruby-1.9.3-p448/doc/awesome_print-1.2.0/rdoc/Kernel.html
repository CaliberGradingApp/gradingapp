def ai(options = {})
  ap = AwesomePrint::Inspector.new(options)
  awesome = ap.awesome self
  if options[:html]
    awesome = "<pre>#{awesome}</pre>"
    awesome = awesome.html_safe if defined? ActiveSupport
  end
  awesome
end
