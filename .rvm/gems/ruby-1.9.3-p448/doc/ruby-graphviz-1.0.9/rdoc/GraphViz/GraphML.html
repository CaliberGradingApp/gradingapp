def initialize( file_or_str )
  data = ((File.file?( file_or_str )) ? File::new(file_or_str) : file_or_str)
  @xmlDoc = REXML::Document::new( data )
  @attributes = {
    :nodes => {},
    :edges => {},
    :graphs => {},
    :graphml => {},
    :endpoint => {},
    :port => {},
    :hyperedge => {}
  }
  @ignored_keys = []
  @graph = nil
  @current_attr = nil
  @current_node = nil
  @current_edge = nil
  @current_graph = nil

  parse( @xmlDoc.root )
end
