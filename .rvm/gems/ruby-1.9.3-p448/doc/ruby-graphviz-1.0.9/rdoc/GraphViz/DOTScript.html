def add_type(type, data)
  return self if data.empty?

  case type
  when "graph_attr"
    append_statement("  " + data)
  when "node_attr"
    append_statement("  node [" + data + "]")
  when "edge_attr"
    append_statement("  edge [" + data + "]")
  else
    raise ArgumentError,
      "Unknown type: #{type}." <<
      "Possible: 'graph_attr','node_attr','edge_attr'"
  end

  self
end
