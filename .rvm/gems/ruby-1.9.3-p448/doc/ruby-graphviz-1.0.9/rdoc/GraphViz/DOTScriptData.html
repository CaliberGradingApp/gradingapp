def to_str
  case @type
    when "graph_attr" then "#{@data.join}#{@separator}"
    when "node_attr" then "node[#{@data.join(' ')}];"
    when "edge_attr" then "edge[#{@data.join(' ')}];"
    else raise ArgumentError, "Wrong type: #{@type}."
  end
end
