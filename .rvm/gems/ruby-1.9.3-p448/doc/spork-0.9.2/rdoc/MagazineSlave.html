def run(argv, stderr, stdout)
  $stdout, $stderr = stdout, stderr
  Spork.exec_each_run
  load @test_framework.helper_file
  @test_framework.run_tests(argv, stderr, stdout)
  puts "  <-- Slave(#{@id_num}) run done!"; stdout.flush
end
