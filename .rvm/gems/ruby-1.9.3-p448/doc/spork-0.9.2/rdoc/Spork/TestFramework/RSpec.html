def run_tests(argv, stderr, stdout)
  if rspec1?
    ::Spec::Runner::CommandLine.run(
      ::Spec::Runner::OptionParser.parse(argv, stderr, stdout)
    )
  else
    ::RSpec::Core::CommandLine.new(argv).run(stderr, stdout)
  end
end
