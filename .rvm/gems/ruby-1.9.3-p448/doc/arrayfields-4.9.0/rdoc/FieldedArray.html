def [](*pairs)
  pairs.flatten!
  raise ArgumentError, "argument must be key/val pairs" unless 
    (pairs.size % 2 == 0)
  fields, elements = [], []
  while((f = pairs.shift) and (e = pairs.shift)) 
    fields << f and elements << e
  end
  new fields, elements
end
